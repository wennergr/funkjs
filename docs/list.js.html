function List() {

  var argToArray = function(arg) {
    var p = Array();
    for(var i=0; i<arg.length; i++) 
      p.push(arg[i]);

    return p
  };

  var build = function(lst) {
    if (lst.length === 0) return NilO;
    else return new Cons(lst[0], build(lst.slice(1)));
  };

  return build(argToArray(arguments));
}
